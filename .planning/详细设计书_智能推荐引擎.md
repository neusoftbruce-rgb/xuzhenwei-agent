### 1.2 完全なソースコード

```java
package com.xuzhenwei.agent.technique;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Service;
import java.util.*;

/**
 * 智能推荐数量决策引擎
 *
 * <p>三层判断架构：
 * <ol>
 *   <li>复杂度分类 — 根据问题关键词判断问题复杂度等级（本地，0ms）</li>
 *   <li>置信度截断 — 在等级上限内，用分数断崖自动截断（本地，0ms）</li>
 *   <li>AI复核 — 模糊场景异步调DeepSeek确认（预留，Phase 4实施）</li>
 * </ol>
 *
 * <p>设计原则：简单问题推1-2条、单一领域推2-4条、复杂问题推3-5条、全面诊断推5-8条。
 * 替换之前硬编码的 .limit(9)。</p>
 *
 * @since 2026-06-17 (v2.1)
 */
@Service
public class RecommendationDecisionEngine {

    private static final Logger log = LoggerFactory.getLogger(RecommendationDecisionEngine.class);

    /** 分数断崖阈值：相邻两条技法分数差超过此值则截断 */
    private static final double CLIFF_THRESHOLD = 0.15;

    /**
     * 复杂度等级
     */
    public enum ComplexityLevel {
        /** 简单事实/定义问题 — 1-2条 */
        SIMPLE_FACTUAL(1, 2),
        /** 单一领域问题 — 2-4条 */
        SINGLE_DOMAIN(2, 4),
        /** 复杂多领域问题 — 3-5条 */
        COMPLEX_MULTI(3, 5),
        /** 全面诊断/战略问题 — 5-8条 */
        COMPREHENSIVE(5, 8);

        final int minRecommend;
        final int maxRecommend;

        ComplexityLevel(int min, int max) {
            this.minRecommend = min;
            this.maxRecommend = max;
        }

        public int getMinRecommend() { return minRecommend; }
        public int getMaxRecommend() { return maxRecommend; }
    }

    /**
     * 【第一层】问题复杂度分类
     *
     * <p>判定ロジック（优先级从高到低）：</p>
     * <ol>
     *   <li>全面诊断/战略关键词 → COMPREHENSIVE</li>
     *   <li>事实/定义询问关键词 → SIMPLE_FACTUAL</li>
     *   <li>多领域交叉（>50字 或 多领域关键词命中）→ COMPLEX_MULTI</li>
     *   <li>默认 → SINGLE_DOMAIN</li>
     * </ol>
     */
    public ComplexityLevel classify(String userInput) {
        if (userInput == null || userInput.isBlank()) {
            return ComplexityLevel.SINGLE_DOMAIN;
        }

        String input = userInput.toLowerCase().trim();
        int length = userInput.length();

        // 1. 全面诊断/战略型关键词 → COMPREHENSIVE
        if (matchesAny(input,
                "全面", "综合诊断", "整体", "战略", "改革", "转型", "重构",
                "全盘", "通盘", "彻底", "系统性")) {
            log.debug("分类=COMPREHENSIVE (战略关键词)");
            return ComplexityLevel.COMPREHENSIVE;
        }
        // 长文本 + 分析关键词 → COMPREHENSIVE
        if (length > 80 && matchesAny(input, "分析", "诊断", "评估", "规划", "设计")) {
            log.debug("分类=COMPREHENSIVE (长文本+分析)");
            return ComplexityLevel.COMPREHENSIVE;
        }

        // 2. 事实/定义询问型关键词 → SIMPLE_FACTUAL
        if (matchesAny(input,
                "是什么", "什么是", "定义", "介绍一下", "介绍", "怎么用",
                "什么意思", "含义", "概念") && length < 40) {
            log.debug("分类=SIMPLE_FACTUAL");
            return ComplexityLevel.SIMPLE_FACTUAL;
        }
        // 单一简单动词问题
        if (matchesAny(input, "你好", "谢谢", "再见", "帮助", "帮助文档") && length < 15) {
            log.debug("分类=SIMPLE_FACTUAL (简单交互)");
            return ComplexityLevel.SIMPLE_FACTUAL;
        }

        // 3. 多领域交叉检测
        int domainCount = countDomainMatches(input);
        if (domainCount >= 2 || length > 50) {
            log.debug("分类=COMPLEX_MULTI (领域数={}, 字数={})", domainCount, length);
            return ComplexityLevel.COMPLEX_MULTI;
        }

        // 4. 默认 → SINGLE_DOMAIN
        log.debug("分类=SINGLE_DOMAIN (默认)");
        return ComplexityLevel.SINGLE_DOMAIN;
    }

    /**
     * 【第二层】置信度截断
     *
     * <p>在复杂度等级上限内，根据分数断崖自动确定截断点。</p>
     *
     * @param sorted 按分数降序排列的匹配结果
     * @param level  复杂度等级
     * @return 截断条数
     */
    public int determineCutoff(List<? extends ScoredItem> sorted, ComplexityLevel level) {
        if (sorted.isEmpty()) return 0;

        int max = level.getMaxRecommend();
        int min = Math.min(level.getMinRecommend(), sorted.size());

        // 如果结果数 <= 最小推荐数，全返回
        if (sorted.size() <= min) return sorted.size();

        // 从 min 条开始，逐个检查分数断崖
        int cutoff = min;
        for (int i = min - 1; i < sorted.size() - 1 && i < max - 1; i++) {
            double gap = sorted.get(i).score() - sorted.get(i + 1).score();
            if (gap > CLIFF_THRESHOLD) {
                // 发现断崖，在此截断
                cutoff = i + 1;
                log.debug("分数断崖截断: 第{}条({:.2f})→第{}条({:.2f}) gap={:.2f}",
                        i + 1, sorted.get(i).score(),
                        i + 2, sorted.get(i + 1).score(), gap);
                break;
            }
            cutoff = i + 2; // 无断崖，继续
        }

        int result = Math.min(cutoff, max);
        log.debug("推荐数量截断结果: {} / {} 条 (等级上限={})", result, sorted.size(), max);
        return result;
    }

    /**
     * 判断是否需要AI复核（预留，Phase 4 实施）
     */
    public boolean needsAiReview(ComplexityLevel level, List<? extends ScoredItem> results) {
        // TODO: Phase 4 — 当分类边界模糊时触发AI验证
        return false;
    }

    // ---- 私有辅助方法 ----

    private boolean matchesAny(String input, String... keywords) {
        for (String kw : keywords) {
            if (input.contains(kw)) return true;
        }
        return false;
    }

    private int countDomainMatches(String input) {
        int count = 0;
        // 农业领域关键词
        if (matchesAny(input, "酵素菌", "堆肥", "微生物", "土壤", "肥料", "病害", "虫害", "农药")) count++;
        // 商业领域关键词
        if (matchesAny(input, "赚钱", "定价", "财务", "成本", "收入", "利润", "商业模式", "获客", "营销", "推广", "销售", "品牌")) count++;
        // 管理领域关键词
        if (matchesAny(input, "团队", "管理", "培训", "招聘", "考核", "绩效", "组织")) count++;
        // 技术/产品领域关键词
        if (matchesAny(input, "产品", "开发", "技术", "种植", "养殖", "加工", "冷链", "包装")) count++;
        // 内容/媒体领域关键词
        if (matchesAny(input, "视频", "短视频", "抖音", "文案", "选题", "自媒体", "内容", "直播")) count++;
        return count;
    }

    /**
     * 分数项接口 — 供 determineCutoff 使用
     */
    public interface ScoredItem {
        double score();
    }
}
```

### 2.2 MatchResult 适配

`MatchResult` 需要实现 `RecommendationDecisionEngine.ScoredItem` 接口：
```java
record MatchResult(String techniqueId, String techniqueName, double score, String reason)
        implements RecommendationDecisionEngine.ScoredItem {
    // score() 方法由 record 自动生成，名称匹配接口的 score()
    // 但 record 生成的是 score() 不是 Score()
    // 需要确认: Java record 的 accessor 名为 score()，与 ScoredItem.score() 一致
}
```

注意：`MatchResult` 是 `record`，其 `double score()` 访问器自动匹配 `ScoredItem.score()` 接口方法。但需要显式声明 `implements RecommendationDecisionEngine.ScoredItem`。
